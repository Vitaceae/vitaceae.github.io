## Create a Package

Example: hello bitbake

  • 參考 Yocto Project: Recipe License Fields

# 編譯期依賴
DEPENDS = "gtk+"

### Select source loction 

# supported protocol: http, ftp, git, svn, file
# 版本資訊 ${P}: ${PN}-${PV}
#   ${PN} 套件名 ${PV} 套件版本
SRC_URI = "http://0.0.0.0/test/${P}.tar.bz2"

# derived from shell command: md5sum $file_name
SRC_URI[md5sum] = "00000000001111111111222222222233"

# derived from shell command: sha256sum $file_name
SRC_URI[sha256sum] = "0000000000111111111122222222223333333333444444444455555555556666"

## Hints

### Find recipes 

$ find -name "*busybox*.bb*"

$ bitbake -e | grep -w $MY_ENV

# e.g. bitbake -e lk | grep ^S=

$ bitbake -e <target> | grep ^WORKDIR=

$ bitbake -c devshell <target>

### Search path 

$ bitbake -e ota_upgrade| grep ^FILESPATH= | sed 's/:/\n/g'

# find cross toolchain path
$ bitbake -c devshell __recipe_name; echo $CC

### Show Debug Information 

$ bitbake -D $target
$ bitbake -DD $target
$ bitbake -DDD $target

## recipe

  • bitbake-layers: 建立 layer; 修改 bblayers.conf; 查看 layer 包含的 recipes
  • recipetool: 建立/修改 recipe; 產生 .bb 及 .bbappend
  • devtool: 開發測試; 不更改設定產生獨立 workspace layer, 與既有 layers 整合測試後新增/更新 layer
    • 目標 layer 需存在於 bblayer.conf 中

### Example: core-image-miniaml 

$ . oe-init-build-env

# 產生必要系統檔
$ bitbake core-image-minimal

# 在 build 建立 layer
$ bitbake-layers create-layer meta-layer

# 將 layer 加入 bblayers.conf
$ bitbake-layers add-layer meta-layer

$ ls

bitbake-cookerdaemon.log  cache  conf  downloads  meta-layer  sstate-cache  tmp

### Generate Recipe

devtool

  • package: GIFLIB
  • dependencies: autotool, configure, Makefile
$ devtool add giflib git://git.code.sf.net/p/giflib/code
$ devtool build giflib
$ devtool build-image core-image-minimal
# 將 giflib recipe 加入 meta-layer
$ devtool finish giflib meta-layer

recipetool

yocto recipe : (1) 撰寫 recipe

複製 Example: recipetool 到 meta-layer/recipes-myproject/myproject/files

NOTE: yocto 預設附檔目錄: files

根據 myproject.bb.mod 修改

$ bitbake myproject

# Add Recipe to Image

修改 build/local.conf:

# 不推薦寫法: IMAGE_INSTALL += " [recipe_basename]"
IMAGE_INSTALL_append = " [recipe_basename]"

僅關聯特定 image 的 recipe:

IMAGE_INSTALL_append_pn-core-image-minimal = " [recipe_basename]"

# .bbappend

.bb 用於描述 recipe, 會因 patch 或 task 不同而需修改, 可使用 .bbappend 擴充而不修改 .bb

使用情境

  1. recipe 已存在於 yocto 基本 meta layer
    • e.g. u-boot_2017.01.bb
  2. layer 共用 recipe

package: giflib(giflib-5.1.4.tar.bz2)

$ . oe-init-build-env
# move 'giflib-5.1.4.tar.bz2' here
$ mkdir 3rd-pkgs; cd 3rd-pkgs/
$ bitbake-layers create-layer meta-custom
$ bitbake-layers add-layer meta-custom
$ cd meta-custom/recipes-example; rm -rf example/; mkdir giflib; cd ../../
$ recipetool create -o meta-custom/recipes-example/giflib/giflib_5.1.4.bb  3rd-pkgs/giflib-5.1.4.tar.bz2

以 devtool 產生 .bbappend 並存放於 meta-custom-devtool-patch layer:

$ bitbake-layers create-layer meta-custom-devtool-patch
$ bitbake-layers add-layer meta-custom-devtool-patch
$ mkdir meta-custom-devtool-patch/recipes-example/giflib

# 使用 devtool 會產生 workspace layer 並複製到 workspace/source
# 以下修改 util/giffix.c

$ devtool modify giflib
NOTE: Source tree extracted to /media/disk1-1/Github/poky/build/workspace/sources/giflib
NOTE: Recipe giflib now set up to build from /media/disk1-1/Github/poky/build/workspace/sources/giflib

int main(int argc, char **argv)
{
    ...
    printf("hello !!! (devtool)\n");
    ...
}

$ devtool finish giflib meta-custom-devtool-patch

手動產生 .bbappend 並存放於 meta-custom-manual-patch layer:

NOTE: yocto 預設搜尋 giflib/giflib, 修改 FILEPATHS(.bb) 或 FILESEXTRAPATHS(.bbappend)

$ bitbake-layers create-layer meta-custom-manual-patch
$ bitbake-layers add-layer meta-custom-manual-patch
$ mkdir meta-custom-manual-patch/recipes-example/giflib/ meta-custom-manual-patch/recipes-example/giflib/giflib

# 手動複製 giflib 並修改 util/giffix.c

$ cd 3rd-pkgs; cp -fa giflib-5.1.4 giflib-5.1.4_modify
$ diff -Naur giflib-5.1.4/util/giffix.c giflib-5.1.4_modify/util/giffix.c > 00_giflib_util_giffix.patch
$ cd ..; mv 3rd-pkgs/00_giflib_util_giffix.patch meta-custom-manual-patch/recipes-example/giflib/giflib/

int main(int argc, char **argv)
{
    ...
    printf("hellow !! (manual)\n");
    ...
}

# 建立 giflib_5.1.4.bbappend
$ vim meta-custom-manual-patch/recipes-example/giflib/giflib_5.1.4.bbappend

FILESEXTRAPATHS_prepend := "${THISDIR}/${PN}:"
SRC_URI_append = "\
    file://00_giflib_util_giffix.patch \

## OE Task Order

build/…/log.task_order

  1. do_setscene
  2. do_fetch
  3. do_unpack
  4. do_patch
  5. do_configure
  6. do_qa_configure
  7. do_compile
  8. do_stage
  9. do_install
  10. do_package
  11. do_populate_staging
  12. do_package_write_deb
  13. do_package_write
  14. do_distribute_sources
  15. do_qa_staging
  16. do_build
  17. do_rebuild

  • do_cleansstate

    • 移除某個對象的所有外部輸出文件以及共享狀態(sstate)緩存內容（不包括遠程鏡像共享內容）。
    • $ bitbake -ccleansstaterecipe

  • do_cleanall

    • 刪除目標對象的所有外部輸出文件，共享狀態緩存，以及下載的源文件（想重新執行do_fetch 過程）。
    • $ bitbake -c cleanallrecipe

  • do_fetch

    • 下載源文件，它用SRC_URI變量以及 不同的變量名前綴來決定適當的下載模式，如調用本地文件： 
        SRC_URI =" \
      file://0001-ccsp-gwprovapp-port-for-puma7-atom.patch;patch=1

  • do_unpack

    • 提取sourcecode中的內容放到${WORKDIR}所指定的工作目錄中。變量S指定了被提取出的sourcecode的最終歸屬位置。

  • do_patch

    • 定位patch文件，並且將他們運用到sourcecode中。

  • do_populate_lic

    • 在構建image的時候爲recipe寫license。

  • do_configure

    • 配置源數據是否打開或者關閉，那些將要被build的軟件的build-time和一些配置選項。do_configure,運行是當前的工作目錄切換到${B}（自行定義的工作目錄）。
    • 此任務的默認行爲是當發現一個makefile的CLEANBROKEN沒有被設置爲“1”，將運行oe_runmake clean。其他情況do_configure 不做任何操作。

  • do_compile

    • 編譯sourcecode，並且運行時也會將當前的工作目錄切換到${B}。
    • 如果發現makefile其默認運行oe_runmake，如果沒有找到,do_compile 不做其他操作。

  • do_install

    • 拷貝將要打包的文件到保持區域${D}。其運行時將當前工作路徑切換至編譯目錄${B}。do_install 以及其他那些直接或者間接依賴與安裝文件的任務（如do_package,do_package_write_*, do_rootfs），都運行在fakeroot的權限下。

  • do_package

    • 分析保持區域${D} 的內容，並根據可獲得的數據包package和文件file，將其內容分割成多個子集。
    • do_package 與do_packagedata 任務相關聯，並且保存了一些重要的元數據。

  • do_packagedata

    • 保存由do_package所產生的源數據到PKGDATA_DIR目錄下，使得內容變成全局變量。

  • 每項任務可添加 _append _prepend 後綴

    • do_configure_prepend ()
    • do_compile_append ()
    • python do_compile_prepend ()

  • 除 python 前的函數都使用 shell 語法定義

  • do_compile 之類函數定義於 openembedded 的 classes; 顯示 oe 基於 bitbake 架構完成

    • meta/poky/meta/classes/kernel.bbclass 
      EXPORT_FUNCTIONS do_compile do_install do_configure

  • 使用 addtask 調整 task 間順序

    • addtask assemble_fitimage before do_install after do_compile

  • MIRRORS_prepend () 修改 .bb 的 mirror

# Function in a Task

need to be prefixed with do_

Function Types:

  1. shell function
do_shell_msg () {
    bbwarn 'hello'
}
  1. BitBake-style Python functions
python do_py_msg () {
    import time
    bb.warn( "hello %s" % ( time.strftime('%Y%m%d', time.gmtime())))
}

# Add/Delete/Insert/Ignore Task 

# Add
addtask task [before task] [after task]
# e.g. 
addtask do_msg before do_configure after do_fetch

# Insert: [task function name]_append() or [task function name]_prepend()
do_msg_prepend () {
    bbwarn 'before'
}
do_msg_append () {
    bbwarn 'before'
}
do_msg () {
    bbwarn 'kkk'
}
addtask do_msg

# Delete: add dependent tasks will be deleted
deltask do_configure

# Ignore
do_task_name[noexec] = "1"

FILESEXTRAPATHS_prepend := "${THISDIR}/files"

Modifying Rootfs Directly:

  • ROOTFS_POSTPROCESS_COMMAND: calls the given functions after bitbake has created the rootfs
  • ROOTFS_POSTINSTALL_COMMAND: calls the given functions after bitbake has installed packages
  • ROOTFS_POSTUNINSTALL_COMMAND: calls the given functions after bitbake has uninstalled packages
  • ROOTFS_PREPROCESS_COMMAND: calls the given functions before OE creates the rootfs
my_function() {
    //
}

ROOTFS_POSTPROCESS_COMMAND += " my_function; "

Adding Packages to SDK

# add to recipe or local.conf
IMAGE_INSTALL_append = " cppzmq-dev"

    nativesdk-zeromq-dev \

## .bb

# Add User Service 

SRC_URI += "file://mai-hald-user.service"
SRC_URI += "file://mai-hald-user.sh"

FILES_${PN} += "${systemd_user_unitdir}"
 
do_install() {
        install -d ${D}${systemd_user_unitdir}/default.target.wants

        install -m 0644 -p -D ${WORKDIR}/mai-hald-user.service ${D}${systemd_user_unitdir}/mai-hald-user.service
        mkdir -p ${D}/etc/systemd/user/default.target.wants/
        ln -sf ${systemd_user_unitdir}/mai-hald-user.service ${D}/etc/systemd/user/default.target.wants/mai-hald-user.service
        install -m 0755 ${WORKDIR}/mai-hald-user.sh ${D}${base_bindir}
}
