CGI, Common Gateway Interface 協定: 外部應用程式與 Web 伺服器溝通的協議。CGI 在伺服器端執行並回傳動態生成的結果。

  1. 將執行檔放在 Web server(IIS, Apache, …) 具執行權限的目錄下
    • Apache 主機參考 httpd.conf 設定, 常見預設值為 /usr/lib/cgi-bin/ 以及 /var/www/cgi.index
    • IIS 主機啟動 IIS Service 並設定目錄權限
  2. html 發送請求透過表單函數 GET/POST
    • GET 的 url 會改變, 能查看參數, 參數以 & 區隔, 無法傳送大量資料且較不安全
    • POST 傳送完整表單, 可傳送大量資料且較安全

# Enable CGI on Apache2 

$ vi /etc/apache2/sites-enabled/000-default.conf  

    ScriptAlias /cgi-bin/ /usr/lib/cgi-bin/
    <Directory "/usr/lib/cgi-bin">
        AllowOverride All
        Options +ExecCGI -MultiViews +SymLinksIfOwnerMatch
        Order allow,deny
        Allow from all
    </Directory>

$ a2enmod cgid
$ sudo systemctl restart apache2

# 編譯後複製到 /usr/lib/cgi-bin/ 目錄下
$ gcc hello.c -o hello
$ sudo mv hello /usr/lib/cgi-bin/

# test on WSL terminal
$ curl localhost/cgi-bin/hello.cgi

# test on browser
# explore http://[my-inet]/cgi-bin/hello.cgi

# Example: 取得環境參數 

#include <stdio.h>

int main(int argc, char* argv[], char *envp[])
{
    printf("Content-type: text/html\n\n");

    for (int i = 0; envp[i] != NULL; ++i) {
        printf(envp[i]);
        printf("<br>");
    } 
}

## GET 

#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    char *data;
    char a[10], b[10];

    printf("Content-Type:text/html\n\n");
    printf("<HTML>\n");
    printf("<HEAD>\n<TITLE >Get Method</TITLE>\n</HEAD>\n");
    printf("<BODY>\n");
    printf("<div style=\"font-size:12px\">\n");
    data = getenv("QUERY_STRING");
    printf("data:%s",data);
    if (sscanf(data,"a=%[^&]&b=%s", a, b) != 2) {
        printf("<DIV STYLE=\"COLOR:RED\">Error: null parameters< /DIV>\n");
    } else {
        printf("<DIV STYLE=\"COLOR:GREEN; font-size:15px;font-weight:bold\">a + b =%d</DIV>\n",atoi(a) + atoi(b));
    }
    printf("<HR COLOR=\"blue\" align=\"left\"width=\"100\">");
    printf("<input type=\"button\" value=\"Back\" onclick=\"javascript:window.location='../index.html'\">");
    printf("</div>\n");
    printf("</BODY>\n");
    printf("</HTML>\n");

    return 0;
}

## POST 

#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    int len;
    char *lenstr, poststr[20];
    char m[10],n[10];

    printf("Content-Type:text/html\n\n");
    printf("<HTML>\n");
    printf("<HEAD>\n<TITLE >post Method</TITLE>\n</HEAD>\n");
    printf("<BODY>\n");
    printf("<div style=\"font-size:12px\">\n");
    lenstr = getenv("CONTENT_LENGTH");
    if (lenstr == NULL) {
        printf("<DIV STYLE=\"COLOR:RED\">Error: null parameters< /DIV>\n");
    } else {
        len = atoi(lenstr);
        fgets(poststr,len+1,stdin);
        printf("%s\n",poststr);
        if (sscanf(poststr,"m=%[^&]&n=%s",m,n) != 2) {
            printf("<DIV STYLE=\"COLOR:RED\">Error: invalid parameters< /DIV>\n");
        } else {
            printf("<DIV STYLE=\"COLOR:GREEN; font-size:15px;font-weight:bold\">m*n=%d</DIV>\n", atoi(m) * atoi(n));
        }
        printf("<HR COLOR=\"blue\" align=\"left\"width=\"100\">");
        printf("<input type=\"button\" value=\"Back\" onclick=\"javascript:window.location='../index.html'\">");
        printf("</div>\n");
        printf("</BODY>\n");
        printf("</HTML>\n");

        fflush(stdout);

        return 0;
    }
}
