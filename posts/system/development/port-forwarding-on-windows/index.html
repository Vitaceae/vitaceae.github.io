ssh 透過端口轉發連線至遠端 Windows 的 WSL

# Configure port-forwarding for SSH service

remote host 將對外 IP 10.20.30.40 的 4000 埠轉發至 WSL 使用的 172.1.2.3:22

# forward 10.20.30.40:4000 to 172.1.2.3:22
> netsh interface portproxy add v4tov4 listenport=4000 listenaddress=0.0.0.0 connectport=22 connectaddress=172.1.2.3

Test connection on local host

> Test-NetConnection 10.20.30.40 -p 4000

# Configure SSH on local host

產生 SSH key

$ ssh-keygen -t ed25519
$ vi ~/.ssh/config
Host            Remote-WSL      #alias of remote host name
HostName        10.20.30.40     #remote host IP
Port            4000            #may be changed
User            mike            #WSL login name on remote host
AddKeysToAgent  yes
IdentitiesOnly  yes
IdentityFile    /c/Users/mike/.ssh/id_ed25519

讓 SSH daemon 自動載入對應密鑰

$ vi ~/.bashrc

SSH_ENV=$HOME/.ssh/environment

function start_agent {
    echo "Initialising new SSH agent..."
    /usr/bin/ssh-agent | sed 's/^echo/#echo/' > "${SSH_ENV}"
    echo succeeded
    chmod 600 "${SSH_ENV}"
    . "${SSH_ENV}" > /dev/null
    /usr/bin/ssh-add;
}
 
# Source SSH settings, if applicable
 
if [ -f "${SSH_ENV}" ]; then
    . ${SSH_ENV} > /dev/null
    #ps ${SSH_AGENT_PID} doesn't work under cygwin
    ps -ef | grep ${SSH_AGENT_PID} | grep ssh-agent$ > /dev/null || {
        start_agent;
    }
else
    start_agent;
fi
 
ssh-add ~/.ssh/id_ed25519

重新啟動 ssh daemon 以套用設定

$ source ~/.bashrc

# Connect to remote WSL via SSH 

# 'Remote-WSL' 即 .ssh/config 中設定的 alias name
# 將 ssh key 綁定 WSL 帳號 mike
$ ssh-copy-id mike@Remote-WSL
# 使用 ssh key 登入
$ ssh Remote-WSL

# Set up SAMBA service on remote host

ignore firewalld settings

$ sudo apt install samba smbclient samba-common

# smb services
$ sudo systemctl start smbd; sudo systemctl enable smbd
$ sudo systemctl start nmbd; sudo systemctl enable nmbd

$ sudo vi /etc/samba/smb.conf

[homes]
    comment = Home Directories
    valid users = %S, %D%w%S
    browseable = yes
    read only = no
    create mask = 0700
    directory mask = 0700
    inherit acls = Yes
    follow symlinks = yes
    wide links = yes

# add new user account to samba database
$ SMB_USER=
$ sudo usermod -a -G smbgrp $SMB_USER
$ sudo smbpasswd -a $SMB_USER

#enter password

$ sudo systemctl restart smbd; sudo systemctl restart nmbd

# configure settings for per-user
$ sudo vi /etc/samba/smb.conf
browseable = no
include = /etc/samba/per-user/%u.conf

$ sudo mkdir /etc/samba/per-user
$ sudo vi /etc/samba/per-user/$SMB_USER.conf
browseable = yes

# test connection
$ smbclient //localhost -U=$SMB_USER

# Set up port-forwarding for SAMBA service on remote host 

# forward 10.20.30.40:4100 to 172.1.2.3:445
> netsh interface portproxy add v4tov4 listenport=4100 listenaddress=0.0.0.0 connectport=445 connectaddress=172.1.2.3

# Mount remote host’s home directory on local WSL 

$ SMB_USER=
$ LINUX_USER=

# test connection
$ smbclient //10.20.30.40/$SMB_USER -U=$SMB_USER --port=4100

# mount remote home directory to /mnt/smb_pc
$ sudo apt install cifs-utils
$ sudo mkdir /mnt/smb_pc
$ sudo mount -t cifs //10.20.30.40/$SMB_USER /mnt/smb_pc -o username=$SMB_USER,password=$PASSWD,port=4100,uid=$LINUX_USER
