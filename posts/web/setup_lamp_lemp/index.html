# LAMP: Install Apache

…

$ sudo apt update && sudo apt upgrade
$ sudo apt install apache2

$ sudo systemctl start apache2
# grant access to .htaccess
$ sudo a2enmod rewrite
$ sudo systemctl restart apache2
$ sudo vi /etc/apache2/sites-enabled/000-default.conf

<Directory /var/www/html>
    AllowOverride All
</Directory>

# LEMP: Install Nginx 

$ sudo apt install nginx
$ sudo systemctl start nginx

# grant access to .htaccess: uncomment "location ~ /\.ht"
$ sudo vi /etc/nginx/sites-available/default

        location ~ /\.ht {
               deny all;
        }

$ sudo systemctl restart nginx

# Install PHP 

# to retrieve packages of newer versions
$ sudo add-apt-repository ppa:ondrej/php
$ sudo apt update && sudo apt upgrade

# LAMP: php7.3, LEMP: php8.1
$ VPHP=php7.3

$ sudo apt install $VPHP
$ sudo apt install php-pear $VPHP-curl $VPHP-dev $VPHP-gd $VPHP-mbstring $VPHP-zip $VPHP-mysql $VPHP-xml $VPHP-fpm
$ sudo apt install $VPHP-intl

# LAMP
$ sudo a2enmod $VPHP
$ sudo systemctl restart apache2

# LEMP
#   1. uncomment and modify the php-fpm version to 8.1
#   2. add index.php
$ sudo vi /etc/nginx/sites-available/default

        location ~ \.php$ {
                include snippets/fastcgi-php.conf;
        #    
        #       # With php-fpm (or other unix sockets):
                fastcgi_pass unix:/var/run/php/php8.1-fpm.sock;
        #       # With php-cgi (or other tcp sockets):
        #       fastcgi_pass 127.0.0.1:9000;
        }

        ...
        # Add index.php to the list if you are using PHP
        index index.php index.html index.htm index.nginx-debian.html;

$ sudo systemctl start php8.1-fpm

## Multi version PHP

Configuring Apache HTTP to Use the FastCGI Process Manager

  • actions: 處理 CGI scripts based on media type or request method.
  • fcgid: 替代 mod_cgi, 處理 concurrent requests 效率更高
  • alias: 映射 host filesystem 目錄或 recirect URL
  • proxy_fcgi: 允許 Apache forward request 給 PHP-FPM
# disable the PHP modules related to mpm_prefork
$ sudo a2dismod php7.3 php8.1
$ sudo a2dismod mpm_prefork
$ sudo systemctl restart apache2

# enable event MPM module
$ sudo a2enmod mpm_event

$ sudo apt install php-fpm libapache2-mod-fcgid
$ sudo a2enmod actions fcgid alias proxy_fcgi
$ sudo a2enconf php7.3-fpm php8.1-fpm
$ sudo systemctl reload apache2

# test configuration
$ cat /etc/apache2/conf-enabled/php7.3-fpm.conf
$ sudo apachectl configtest
$ sudo systemctl restart apache2

# check if mpm, proxy are active
# sudo apachectl -M | grep 'proxy'

建立使用不同 PHP 版本的測試站台

$ sudo mkdir /var/www/site1
$ sudo chown -R www-data:www-data /var/www/site1
$ sudo chmod -R 755 /var/www/site1

$ sudo vi /var/www/site1/info.php

<?php phpinfo(); ?>

$ sudo vi /etc/apache2/sites-available/site1.conf

<VirtualHost *:8081>
     ServerAdmin admin@localhost
     DocumentRoot /var/www/site1
     DirectoryIndex info.php

     <Directory /var/www/site1>
        Options Indexes FollowSymLinks MultiViews
        AllowOverride All
        Order allow,deny
        allow from all
     </Directory>

    <FilesMatch \.php$>
      SetHandler "proxy:unix:/run/php/php7.3-fpm.sock|fcgi://localhost"
    </FilesMatch>

     ErrorLog ${APACHE_LOG_DIR}/site1.error.log
     CustomLog ${APACHE_LOG_DIR}/site1.access.log combined
</VirtualHost>

$ sudo apachectl configtest
$ sudo a2ensite site1

# listen to the new ports
$ sudo vi /etc/apache2/ports.conf

Listen 8081

$ sudo systemctl reload apache2
$ sudo systemctl daemon-reload

# disable the default site
#$ sudo a2dissite 000-default.conf
#$ sudo systemctl restart apache2

# Install phpmyadmin

LAMP

$ sudo apt-get install phpmyadmin
$ sudo ln -s /usr/share/phpmyadmin /var/www/html/phpmyadmin

LEMP

$ PMA_VER=5.0.2
$ PMA_ARCHIVE=phpMyAdmin-$PMA_VER-all-languages
$ wget https://files.phpmyadmin.net/phpMyAdmin/$PMA_VER/$PMA_ARCHIVE.zip
$ unzip $PMA_ARCHIVE.zip
$ mv $PMA_ARCHIVE pma

# change the salt value
$ cp pma/config.sample.inc.php pma/config.inc.php
$ sudo vi config.inc.php

$cfg['blowfish_secret'] = '****'; /* YOU MUST FILL IN THIS FOR COOKIE AUTH */

$ sudo ln -s ~/pma /var/www/html/pma/

# Run Apache and Nginx simultaneously

Nginx 做為前端: 擅於處理 concurrent requests, 適合 Proxy 任務, 佔用較少系統資源
Apache 做為後端: 處理 Concurrent Queue 時效能優於 FastCGI(Nginx 需要類似方式執行 php); mod_php 處理動態 PHP 頁面時也比 php-cgi 穩定且效率高
故使用 Nginx 接收 client request, 除了動態頁面交由 Apache 處理再透過 Nginx 回傳, 其餘自己處理; 或使用 Nginx 做為 Proxy, 請求都由 Apache, Tomcat 處理

$ sudo vi /etc/apache2/ports.conf

- Listen 80
+ Listen 8080

$ sudo vi /etc/nginx/sites-available/default

    listen 80 default_server;
