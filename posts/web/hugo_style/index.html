# global-wise style 

<cust>
affected by <code>cust.scss</code> and <code>cust.css</code>
</cust>

<div class="cust-style">
example of <code>cust-style</code>
</div>
affected by cust.scss and cust.css
example of cust-style

add the following line to .toml

customCSS = ["custom-css/cust.scss", "custom-css/cust.css"]

add assets/custom-css/cust.css

.cust-style {
    background-color: pink;
    color: black;
    padding: 1em;
    border-radius: 0.3em;
    border: 1px solid black;
}

add assets/custom-css/cust.scss

@import 'foundation/vars';

cust {
    display: block;
    font: 100% $font-stack;
    color: $primary-color;
    background-color: $background-color;
    margin-top: 1em;
}

add assets/custom-css/foundation/_vars.scss

$font-stack: Helvetica, sans-serif;
$primary-color: rgb(255, 255, 168);
$background-color: rgb(85, 85, 85);

# page-wise style

the current page is affected only by hugo-style/style.css

add hugo-style/style.css

.cust-style2 {
    max-width: none;
    padding-left: 5em;
    padding-right: 5em;
    background-color: yellow;
}

example:

<style>
  @import 'style.css';
</style>
